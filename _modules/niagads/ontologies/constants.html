def annotation_property_types():
    pTypes = list(ANNOTATION_PROPERTIES.keys())
    pTypes.remove('db_ref')
    pTypes.remove('synonym')
    return pTypes



ORDERED_PROPERTY_LABELS = ['id', 'iri', 'term'] + annotation_property_types()
